&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Support Vector Machine (SVM)&lt;/span&gt;

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Importing the libraries&lt;/span&gt;
&lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; numpy &lt;span style=&#34;color: #fbde2d;&#34;&gt;as&lt;/span&gt; np
&lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; pandas &lt;span style=&#34;color: #fbde2d;&#34;&gt;as&lt;/span&gt; pd
&lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; matplotlib.pyplot &lt;span style=&#34;color: #fbde2d;&#34;&gt;as&lt;/span&gt; plt

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Importing the dataset&lt;/span&gt;
dataset &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; pd.read_csv(&lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;Social_Network_Ads.csv&#39;&lt;/span&gt;)
features &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; dataset.iloc[:, [&lt;span style=&#34;color: #d8fa3c;&#34;&gt;2&lt;/span&gt;, &lt;span style=&#34;color: #d8fa3c;&#34;&gt;3&lt;/span&gt;]].values
labels &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; dataset.iloc[:, &lt;span style=&#34;color: #d8fa3c;&#34;&gt;4&lt;/span&gt;].values

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Splitting the dataset into the Training set and Test set&lt;/span&gt;
&lt;span style=&#34;color: #fbde2d;&#34;&gt;from&lt;/span&gt; sklearn.model_selection &lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; train_test_split
features_train, features_test, labels_train, labels_test &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; train_test_split(features, labels, test_size &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0.25&lt;/span&gt;, random_state &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0&lt;/span&gt;)

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Feature Scaling&lt;/span&gt;
&lt;span style=&#34;color: #fbde2d;&#34;&gt;from&lt;/span&gt; sklearn.preprocessing &lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; StandardScaler
sc &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; StandardScaler()
features_train &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; sc.fit_transform(features_train)
features_test &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; sc.transform(features_test)

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Fitting SVM to the Training set&lt;/span&gt;
&lt;span style=&#34;color: #fbde2d;&#34;&gt;from&lt;/span&gt; sklearn.svm &lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; SVC
classifier &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; SVC(kernel &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; &lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;linear&#39;&lt;/span&gt;, random_state &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0&lt;/span&gt;)
classifier.fit(features_train, labels_train)

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Predicting the Test set results&lt;/span&gt;
labels_pred &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; classifier.predict(features_test)

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Making the Confusion Matrix&lt;/span&gt;
&lt;span style=&#34;color: #fbde2d;&#34;&gt;from&lt;/span&gt; sklearn.metrics &lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; confusion_matrix
cm &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; confusion_matrix(labels_test, labels_pred)

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Model Score&lt;/span&gt;
score &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; classifier.score(features_test,labels_test)

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Visualising the Training set results&lt;/span&gt;
&lt;span style=&#34;color: #fbde2d;&#34;&gt;from&lt;/span&gt; matplotlib.colors &lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; ListedColormap
features_set, labels_set &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; features_train, labels_train
X1, X2 &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; np.meshgrid(np.arange(start &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; features_set[:, &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0&lt;/span&gt;].min() &lt;span style=&#34;color: #fbde2d;&#34;&gt;-&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;1&lt;/span&gt;, stop &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; features_set[:, &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0&lt;/span&gt;].max() &lt;span style=&#34;color: #fbde2d;&#34;&gt;+&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;1&lt;/span&gt;, step &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0.01&lt;/span&gt;),
                     np.arange(start &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; features_set[:, &lt;span style=&#34;color: #d8fa3c;&#34;&gt;1&lt;/span&gt;].min() &lt;span style=&#34;color: #fbde2d;&#34;&gt;-&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;1&lt;/span&gt;, stop &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; features_set[:, &lt;span style=&#34;color: #d8fa3c;&#34;&gt;1&lt;/span&gt;].max() &lt;span style=&#34;color: #fbde2d;&#34;&gt;+&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;1&lt;/span&gt;, step &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0.01&lt;/span&gt;))
plt.contourf(X1, X2, classifier.predict(np.array([X1.ravel(), X2.ravel()]).T).reshape(X1.shape),
             alpha &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0.75&lt;/span&gt;, cmap &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; ListedColormap((&lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;red&#39;&lt;/span&gt;, &lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;green&#39;&lt;/span&gt;)))
plt.xlim(X1.min(), X1.max())
plt.ylim(X2.min(), X2.max())
&lt;span style=&#34;color: #fbde2d;&#34;&gt;for&lt;/span&gt; i, j &lt;span style=&#34;color: #fbde2d;&#34;&gt;in&lt;/span&gt; &lt;span style=&#34;color: #8da6ce;&#34;&gt;enumerate&lt;/span&gt;(np.unique(labels_set)):
    plt.scatter(features_set[labels_set &lt;span style=&#34;color: #fbde2d;&#34;&gt;==&lt;/span&gt; j, &lt;span style=&#34;color: #d8fa3c;&#34;&gt;0&lt;/span&gt;], features_set[labels_set &lt;span style=&#34;color: #fbde2d;&#34;&gt;==&lt;/span&gt; j, &lt;span style=&#34;color: #d8fa3c;&#34;&gt;1&lt;/span&gt;],
                c &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; ListedColormap((&lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;red&#39;&lt;/span&gt;, &lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;green&#39;&lt;/span&gt;))(i), label &lt;span style=&#34;color: #fbde2d;&#34;&gt;=&lt;/span&gt; j)
plt.title(&lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;SVM (Training set)&#39;&lt;/span&gt;)
plt.xlabel(&lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;Age&#39;&lt;/span&gt;)
plt.ylabel(&lt;span style=&#34;color: #61ce3c;&#34;&gt;&#39;Estimated Salary&#39;&lt;/span&gt;)
plt.legend()
plt.show()

&lt;span style=&#34;color: #aeaeae;&#34;&gt;# Visualising the Test set results&lt;/span&gt;
&lt;span style=&#34;color: #fbde2d;&#34;&gt;from&lt;/span&gt; matplotlib.colors &lt;span style=&#34;color: #fbde2d;&#34;&gt;import&lt;/span&gt; ListedColormap
